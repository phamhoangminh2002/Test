Create database

  • Create database not parameter
    •                     
                            CREATE DATABASE NewDataBase
                        go
  • Create database 2 parameter: primary and log
    •                     CREATE DATABASE NewDataBase2
                    ON PRIMARY
                    (NAME = 'NewDataBase2', FILENAME = 'C:\DATA\NewDataBase2.mdf', 
                    SIZE = 10MB, MAXSIZE = 200MB, FILEGROWTH = 2MB)
                    LOG ON
                    (NAME = 'NewDataBase2_log', FILENAME = 'C:\DATA\NewDataBase2.ldf', 
                    SIZE = 10MB, MAXSIZE = unlimited, FILEGROWTH = 2MB)
                    GO
  • Create database 2 parameter: primary, filegroup and log
    •                     CREATE DATABASE NewDataBase3
                    ON PRIMARY
                    (NAME = 'NewDataBase3', FILENAME = 'C:\DATA\NewDataBase3.mdf',
                     SIZE = 10MB, MAXSIZE = 200MB, FILEGROWTH = 2MB),
                    FILEGROUP myFileGroup
                    (NAME = 'NewDataBase3_FG', FILENAME = 'C:\DATA\NewDataBase3_FG.ndf', SIZE = 10MB, 
                    MAXSIZE = 200MB, FILEGROWTH = 2MB)
                    LOG ON
                    (NAME = 'NewDataBase3_log', FILENAME = 'C:\DATA\NewDataBase3.ldf', 
                    SIZE = 10MB, MAXSIZE = unlimited, FILEGROWTH = 2MB)
                    GO
  • Change database
    •                     ALTER DATABASE NewDataBase
                    MODIFY NAME = NewDataBase1
                    GO
  • Change owner
    •                     EXEC sp_changedbowner 'sa' --'sa' is permission adminTrator
                    GO
  • Drop database
    •                     DROP DATABASE NewDataBase1
                    GO
